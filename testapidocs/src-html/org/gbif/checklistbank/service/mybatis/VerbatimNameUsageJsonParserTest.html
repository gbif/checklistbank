001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.VerbatimNameUsage;
004import org.gbif.api.vocabulary.Extension;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009
010public class VerbatimNameUsageJsonParserTest {
011
012  @Test
013  public void testJsonToVerbatim() throws Exception {
014    final String json = "{\"id\":\"100\",\n" + " \"taxonomicStatus\":\"valid\",\n" + " \"taxonRank\":\"Species\",\n"
015                        + " \"scientificNameAuthorship\":null,\n" + " \"parentNameUsageID\":\"86\",\n"
016                        + " \"acceptedNameUsageID\":null,\n" + " \"scientificName\":\"Spirillum beijerinckii\",\n"
017                        + " \"extensions\": {\n"
018                        + "   \"VernacularName\" : [{\"vernacularName\":\"Strand\", \"language\":\"en\"}, {\"vernacularName\":\"BÃ¶nse\", \"language\":\"se\"}],\n"
019                        + "   \"Distribution\" : [{\"hi\":\"ho\"}]\n" + " }}";
020    VerbatimNameUsageJsonParser parser = new VerbatimNameUsageJsonParser();
021    VerbatimNameUsage v = parser.jsonToVerbatim(json);
022
023    assertEquals("100", v.getFields().get("id"));
024    assertEquals("valid", v.getFields().get("taxonomicStatus"));
025    assertEquals("Spirillum beijerinckii", v.getFields().get("scientificName"));
026    assertEquals("86", v.getFields().get("parentNameUsageID"));
027    assertEquals(2, v.getExtensions().get(Extension.VERNACULAR_NAME).size());
028    assertEquals(1, v.getExtensions().get(Extension.DISTRIBUTION).size());
029  }
030}
