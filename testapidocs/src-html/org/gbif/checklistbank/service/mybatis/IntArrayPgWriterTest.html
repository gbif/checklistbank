001package org.gbif.checklistbank.service.mybatis;
002
003import java.util.List;
004
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertTrue;
009
010public class IntArrayPgWriterTest {
011
012  @Test
013  public void testWrite() throws Exception {
014    assertInput("12\n54321\n23456789\n345670981");
015    assertInput("12\n54321\n23456789\n345670981\n");
016  }
017
018  private void assertInput(final String input) throws Exception {
019    IntArrayPgWriter writer = new IntArrayPgWriter();
020    writer.write(input.toCharArray());
021    List<Integer> result = writer.result();
022    assertEquals(4, result.size());
023    assertTrue(result.contains(12));
024    assertTrue(result.contains(54321));
025    assertTrue(result.contains(23456789));
026    assertTrue(result.contains(345670981));
027  }
028}
