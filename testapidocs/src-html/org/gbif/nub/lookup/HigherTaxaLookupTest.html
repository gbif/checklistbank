014package org.gbif.nub.lookup;
015
016import org.gbif.api.vocabulary.Rank;
017
018import java.io.IOException;
019
020import org.junit.Test;
021
022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertFalse;
024import static org.junit.Assert.assertNull;
025import static org.junit.Assert.assertTrue;
026
027/**
028 * @author markus
029 */
030public class HigherTaxaLookupTest {
031
032  @Test
033  public void testReloading() throws IOException {
034    HigherTaxaLookup htl = new HigherTaxaLookup();
035    htl.loadClasspathDicts("dicts");
036    assertTrue(htl.size() > 10);
037  }
038
039  @Test
040  public void testKingdoms() throws IOException {
041    HigherTaxaLookup htl = new HigherTaxaLookup();
042    htl.loadClasspathDicts("dicts");
043    // Animalia varieties
044    assertEquals("Animalia", htl.lookup("Animalia", Rank.KINGDOM));
045    assertEquals("Animalia", htl.lookup("Anamalia", Rank.KINGDOM));
046    assertEquals("Animalia", htl.lookup("Animal", Rank.KINGDOM));
047    assertEquals("Animalia", htl.lookup("Metazoa", Rank.KINGDOM));
048
049    assertNull(htl.lookup("Incertae sedis", Rank.KINGDOM));
050  }
051
052  @Test
053  public void testBlacklist() throws IOException {
054    HigherTaxaLookup htl = new HigherTaxaLookup();
055    htl.loadClasspathDicts("dicts");
056
057    assertFalse(htl.isBlacklisted("Animals"));
058    assertFalse(htl.isBlacklisted("Abies indeterminata"));
059    assertTrue(htl.isBlacklisted("Unknown"));
060    assertTrue(htl.isBlacklisted("Incertae sedis"));
061  }
062
063  @Test
064  public void testNormalization() throws IOException {
065    assertEquals("ANIMALS", HigherTaxaLookup.norm("Animals"));
066    assertEquals("ANIMALS", HigherTaxaLookup.norm("  Animals"));
067    assertEquals("ANIMALS", HigherTaxaLookup.norm("12Animals???"));
068    assertEquals("DRECKS ASTERACEAE", HigherTaxaLookup.norm("drecks Asteraceae"));
069    assertNull(HigherTaxaLookup.norm(null));
070    assertNull(HigherTaxaLookup.norm(" "));
071    assertNull(HigherTaxaLookup.norm("321"));
072    assertNull(HigherTaxaLookup.norm(""));
073    assertNull(HigherTaxaLookup.norm(",.-öä? "));
074  }
075
076}
