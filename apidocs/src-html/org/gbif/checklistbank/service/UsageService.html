001package org.gbif.checklistbank.service;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.checklistbank.service.mybatis.model.Usage;
007
008import java.util.List;
009import java.util.UUID;
010
011/**
012 * Persistence service dealing with simple reference citations.
013 * This interface is restricted to the mybatis module only!
014 */
015public interface UsageService {
016
017  /**
018   * @return int array of all name usage ids in checklist bank
019   */
020  List<Integer> listAll();
021
022  /**
023   * @return the highest usageKey used in the dataset
024   */
025  Integer maxUsageKey(UUID datasetKey);
026
027  /**
028   * Lists all name usages with a key between start / end.
029   */
030  List<NameUsage> listRange(int usageKeyStart, int usageKeyEnd);
031
032  /**
033   * Page through all usages in a dataset.
034   */
035  PagingResponse<Usage> list(UUID datasetKey, Pageable page);
036
037
038}
