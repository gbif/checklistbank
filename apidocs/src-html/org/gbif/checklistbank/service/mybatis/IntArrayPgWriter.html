001package org.gbif.checklistbank.service.mybatis;
002
003import java.io.IOException;
004import java.io.Writer;
005import java.nio.CharBuffer;
006import java.util.ArrayList;
007import java.util.List;
008
009import com.google.common.collect.Lists;
010
011/**
012 * A writer for native postgres copy commands that converts a pg query result of integers into a integer list
013 * very efficiently.
014 */
015public class IntArrayPgWriter extends Writer {
016  private ArrayList<Integer> array = Lists.newArrayListWithCapacity(10000);
017  private final CharBuffer value = CharBuffer.allocate(16);
018
019  @Override
020  public void close() throws IOException {
021    // nothing to do
022  }
023
024  @Override
025  public void flush() throws IOException {
026    // nothing to do
027  }
028
029  @Override
030  public void write(char[] cbuf, int off, int len) throws IOException {
031    CharBuffer buf = CharBuffer.wrap(cbuf, off, len);
032    while (buf.hasRemaining()) {
033      char c = buf.get();
034      if (c == '\n') {
035        add();
036      } else {
037        value.append(c);
038      }
039    }
040    add();
041  }
042
043
044  private void add() throws IOException {
045    value.flip();
046    if (value.hasRemaining()) {
047      try {
048        array.add(Integer.valueOf(value.toString()));
049      } catch (NumberFormatException e) {
050      } catch (NullPointerException e) {
051      }
052    }
053    value.clear();
054  }
055
056
057  public List<Integer> result() {
058    return array;
059  }
060}
