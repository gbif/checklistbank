001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.checklistbank.service.DatasetImportService;
004import org.gbif.checklistbank.service.mybatis.model.Usage;
005
006import java.util.Iterator;
007import java.util.List;
008import java.util.Map;
009import java.util.UUID;
010import javax.sql.DataSource;
011
012import com.google.common.collect.Lists;
013import com.google.inject.Inject;
014import org.apache.ibatis.session.ExecutorType;
015import org.apache.ibatis.session.TransactionIsolationLevel;
016import org.mybatis.guice.transactional.Transactional;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020/**
021 * Implements the NameUsageService using MyBatis.
022 * All PagingResponses will not have the count set as it can be too costly sometimes.
023 */
024public class DatasetImportServiceMyBatis implements DatasetImportService {
025
026  private static final Logger LOG = LoggerFactory.getLogger(DatasetImportServiceMyBatis.class);
027
028  private final NameUsageMapper mapper;
029  private final UsageMapper usageMapper;
030  private final NubRelMapper nubRelMapper;
031  private final ParsedNameMapper parsedNameMapper;
032
033  @Inject
034  private DataSource ds;
035
036  @Inject
037  DatasetImportServiceMyBatis(NameUsageMapper mapper, UsageMapper usageMapper, ParsedNameMapper parsedNameMapper,
038    NubRelMapper nubRelMapper) {
039    this.mapper = mapper;
040    this.parsedNameMapper = parsedNameMapper;
041    this.usageMapper = usageMapper;
042    this.nubRelMapper = nubRelMapper;
043  }
044
045  @Override
046  public void insertUsages(UUID datasetKey, Iterator<Usage> iter) {
047    final int BATCH_SIZE = 1000;
048
049    int batchCounter = 1;
050    List<Usage> batch = Lists.newArrayList();
051    while (iter.hasNext()) {
052      batch.add(iter.next());
053      if (batch.size() % BATCH_SIZE == 0) {
054        LOG.debug("Insert nub usage batch {}", batchCounter);
055        insertUsageBatch(datasetKey, batch);
056        batchCounter++;
057        batch.clear();
058      }
059    }
060    LOG.debug("Insert last nub usage batch {}", batchCounter);
061    insertUsageBatch(datasetKey, batch);
062  }
063
064  @Transactional(
065      executorType = ExecutorType.BATCH,
066      isolationLevel = TransactionIsolationLevel.READ_UNCOMMITTED,
067      exceptionMessage = "Something went wrong while inserting nub relations batch for dataset {0}"
068  )
069  @Override
070  public void insertNubRelations(UUID datasetKey, Map<Integer, Integer> relations) {
071    nubRelMapper.deleteByDataset(datasetKey);
072    for (Map.Entry<Integer, Integer> entry : relations.entrySet()) {
073      nubRelMapper.insert(datasetKey, entry.getKey(), entry.getValue());
074    }
075  }
076
077  @Transactional(
078      executorType = ExecutorType.BATCH,
079      isolationLevel = TransactionIsolationLevel.READ_UNCOMMITTED,
080      exceptionMessage = "Something went wrong while inserting usage batch for dataset {0}"
081  )
082  private void insertUsageBatch(UUID datasetKey, List<Usage> usages) {
083    for (Usage u : usages) {
084      usageMapper.insert(datasetKey, u);
085    }
086  }
087
088  @Override
089  public void deleteDataset(UUID datasetKey) {
090    LOG.info("Deleting entire dataset {}", datasetKey);
091    usageMapper.deleteByDataset(datasetKey);
092  }
093
094
095}
