001package org.gbif.checklistbank.service;
002
003import org.gbif.api.model.checklistbank.ParsedName;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006
007import java.util.UUID;
008
009/**
010 * Persistence service dealing with parsed names.
011 * This interface is restricted to the mybatis module only!
012 */
013public interface ParsedNameService {
014
015  /**
016   * Get a parsed name by its name key.
017   * @param key the key of the parsed name
018   * @return the parsed name or null
019   */
020  ParsedName get(int key);
021
022  /**
023   * Returns the full parsed name for a given scientific name string.
024   * Inserts any non existing names into the name_string table.
025   */
026  ParsedName createOrGet(String scientificName);
027
028  /**
029   * Page through all names in a dataset.
030   */
031  PagingResponse<ParsedName> listNames(UUID datasetKey, Pageable page);
032
033}
