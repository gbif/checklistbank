001package org.gbif.checklistbank.ws.client;
002
003import org.gbif.api.model.checklistbank.DatasetMetrics;
004import org.gbif.api.service.checklistbank.DatasetMetricsService;
005import org.gbif.checklistbank.ws.client.guice.ChecklistBankWs;
006import org.gbif.checklistbank.ws.util.Constants;
007import org.gbif.ws.client.BaseWsGetClient;
008
009import java.util.List;
010import java.util.UUID;
011import javax.annotation.Nullable;
012
013import com.google.inject.Inject;
014import com.sun.jersey.api.client.GenericType;
015import com.sun.jersey.api.client.WebResource;
016
017/**
018 * Client-side implementation to the DatasetMetricsService.
019 */
020public class DatasetMetricsWsClient extends BaseWsGetClient<DatasetMetrics, UUID> implements DatasetMetricsService {
021
022  private final GenericType<List<DatasetMetrics>> listType = new GenericType<List<DatasetMetrics>>() {};
023
024  @Inject
025  public DatasetMetricsWsClient(@ChecklistBankWs WebResource resource) {
026    super(DatasetMetrics.class, resource.path(Constants.DATASET_METRICS_PATH), null);
027  }
028
029
030  /**
031   * Gets a DatasetMetrics by its dataset key.
032   *
033   * @param datasetKey key of dataset to get the metrics for
034   *
035   * @return requested DatasetMetrics or null if none could be found
036   */
037  @Override
038  public DatasetMetrics get(UUID datasetKey) {
039    return get(datasetKey.toString());
040  }
041
042  @Nullable
043  @Override
044  public List<DatasetMetrics> list(UUID datasetKey) {
045    return get(listType, datasetKey.toString(), "history");
046  }
047}
