001package org.gbif.checklistbank.ws.nub;
002
003
004import org.gbif.checklistbank.service.mybatis.guice.ChecklistBankServiceMyBatisModule;
005import org.gbif.nub.lookup.NubMatchingModule;
006import org.gbif.ws.server.guice.GbifServletListener;
007
008import java.util.List;
009import java.util.Properties;
010
011import com.google.common.collect.Lists;
012import com.google.inject.Module;
013
014public class NubWsListener extends GbifServletListener {
015
016  public NubWsListener() {
017    super("checklistbank.properties", NubResource.class.getPackage().getName(), false);
018  }
019
020  @Override
021  protected List<Module> getModules(Properties properties) {
022    List<Module> modules = Lists.newArrayList();
023    modules.add(new ChecklistBankServiceMyBatisModule(properties));
024    modules.add(new NubMatchingModule());
025    // use the line below to run the webservice locally with the json test index data from the nub module
026    // modules.add(new NubMatchingTestModule());
027    return modules;
028  }
029
030}
