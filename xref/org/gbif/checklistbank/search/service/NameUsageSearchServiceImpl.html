38  /**
39   * Checklist Bank Search interface implementation using Solr/lucene.
40   * This class uses a remote to interact with CLB index.
41   */
42  public class NameUsageSearchServiceImpl
43    extends SolrSearchSuggestService<NameUsageSearchResult, NameUsageSearchParameter, NameUsageSolrSearchResult,
44    NameUsageSearchRequest, NameUsageSuggestRequest> implements NameUsageSearchService {
45  
46    // Default primary sort order
47    private static final Map<String, SolrQuery.ORDER> PRIMARY_SORT_ORDER =
48      new ImmutableMap.Builder<String, SolrQuery.ORDER>()
49        .put("score", SolrQuery.ORDER.desc)
50        .put("num_descendants", SolrQuery.ORDER.desc)
51        .put("scientific_name", SolrQuery.ORDER.asc).build();
52  
53    // Default sort order for suggest results
54    private static final Map<String, SolrQuery.ORDER> DEFAULT_SUGGEST_ORDER =
55      new ImmutableMap.Builder<String, SolrQuery.ORDER>()
56        .put("score", SolrQuery.ORDER.desc)
57        .put("rank_key", SolrQuery.ORDER.asc).build();
58  
59  
60    private static final Multimap<NameUsageSearchParameter, String> SUGGEST_DEFAULT_FILTERS = ArrayListMultimap.create();
61  
62    protected static final int DEFAULT_LIMIT = 10;
63  
64    static {
65      // Suggest default filters for suggest service: names of type INFORMAL and BLACKLISTED are excluded by default.
66      SUGGEST_DEFAULT_FILTERS.put(NameUsageSearchParameter.NAME_TYPE, NOT_OP + NameType.INFORMAL.name());
67      SUGGEST_DEFAULT_FILTERS.put(NameUsageSearchParameter.NAME_TYPE, NOT_OP + NameType.BLACKLISTED.name());
68    }
69  
70    /**
71     * Default constructor.
72     */
73    @Inject
74    public NameUsageSearchServiceImpl(SolrServer server) {
75      // Type parameter bounded to NameUsageSearch
76      super(server, null, NameUsageSearchResult.class, NameUsageSolrSearchResult.class,
77        NameUsageSearchParameter.class, PRIMARY_SORT_ORDER, DEFAULT_SUGGEST_ORDER, false);
78    }
79  
80    @Override
81    public List<NameUsageSearchResult> suggest(NameUsageSuggestRequest suggestRequest) {
82      if (!suggestRequest.getParameters().containsKey(NameUsageSearchParameter.DATASET_KEY)) {
83        // if the datasetKey parameters is not in the list, the GBIF nub is used by default
84        suggestRequest.addParameter(NameUsageSearchParameter.DATASET_KEY, Constants.NUB_DATASET_KEY.toString());
85        for (NameUsageSearchParameter p : SUGGEST_DEFAULT_FILTERS.keys()) {
86          suggestRequest.addParameter(p, SUGGEST_DEFAULT_FILTERS.get(p));
87        }
88      }
89      return super.suggest(suggestRequest);
90    }
