42  /**
43   * Main class to run a full import of CoL GSD annotations for the GBIF Backbone, populating the col_annotation table
44   * in Checklist Bank.
45   * @see <a href="http://dev.gbif.org/issues/browse/CLB-248">CLB-248</a>
46   */
47  public class ColAnnotationImport {
48    private static final Logger LOG = LoggerFactory.getLogger(ColAnnotationImport.class);
49    private Properties props;
50    private ColAnnotationService annotationService;
51    @VisibleForTesting
52    protected Pattern ANNOTATION_SPLITTER = Pattern.compile("^\\s*(.*)\\s*\\|\\s*(.*)\\s*\\|\\s*(.*)\\s*$");

54    public ColAnnotationImport(String propsFile) throws IOException {
55      props = PropertiesUtil.readFromFile(propsFile);
56      // init guice
57      ChecklistBankServiceMyBatisModule myBatisModule = new ChecklistBankServiceMyBatisModule(props);
58      Injector inj = Guice.createInjector(myBatisModule);
59      annotationService = inj.getInstance(ColAnnotationService.class);
60    }

62    public void importAnnotations() throws IOException, URISyntaxException {
63      Archive arch = downloadArchive();
64      int counter = 0;
65      int failures = 0;
66      int empty = 0;
67      for (StarRecord star : arch) {
68        Record rec = star.core();
69        counter++;
70        try {
71          int nubKey = counter;//TODO: use taxonIDs once available: Integer.parseInt(rec.value(DwcTerm.taxonID));
72          Matcher m = ANNOTATION_SPLITTER.matcher(rec.value(DwcTerm.taxonRemarks));
73          if (m.find()) {
74            ColAnnotation annotation = new ColAnnotation(nubKey, rec.value(DwcTerm.datasetName), rec.value(DwcTerm.scientificName),
75              isRejected(m.group(1)), m.group(3), m.group(2));
76            annotationService.insertAnnotation(annotation);
77  
78          } else {
79            empty++;
80            LOG.warn("No annotation given for {}", rec.value(DwcTerm.scientificName));
81          }
82        } catch (NumberFormatException e) {
83          failures++;
84          LOG.warn("No valid taxonID given for {}", rec.value(DwcTerm.scientificName));
85        }
86      }
87      LOG.info("{} annotations processed, {} lack notes, {} failed to import", counter, empty, failures);
88    }
89  
90    @VisibleForTesting
91    protected static boolean isRejected(String annotation) {
92      if (annotation.toLowerCase().startsWith("placed")) {
93        return false;
94      }
95      return true;
96    }

98    private Archive downloadArchive() throws IOException, URISyntaxException {
99      // download url
100     final String url = props.getProperty("col.annotation.url");
101     // local work dir
102     File workDir = FileUtils.createTempDir();
103     workDir.deleteOnExit();
104     // local zip file
105     File zip = new File(workDir, "annotation.zip");
106     // local decompressed dwca
107     File dwca = new File(workDir, "annotation");
108     // create folders
109     org.apache.commons.io.FileUtils.forceMkdir(dwca);
110 
111     // use a 10 minutes timeout
112     HttpClient client = HttpUtil.newMultithreadedClient(600000, 10, 10);
113     // authentication
114     HttpContext authContext = new BasicHttpContext();
115     URI authUri = new URI(url);
116     AuthScope scope = new AuthScope(authUri.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM);
117 
118     CredentialsProvider credsProvider = new BasicCredentialsProvider();
119     credsProvider.setCredentials(scope, new UsernamePasswordCredentials(props.getProperty("col.annotation.user"), props.getProperty("col.annotation.password")));
120     authContext.setAttribute(ClientContext.CREDS_PROVIDER, credsProvider);
121 
122     HttpGet get = new HttpGet(url);
123     HttpResponse response = client.execute(get, authContext);
124 
125     if (response.getStatusLine().getStatusCode() != 200) {
126       LOG.error("{} error downloading annotations from {}: {}", response.getStatusLine(), url,
127         response.getStatusLine().getReasonPhrase());
128       System.exit(1);
129     }
130 
131     HttpEntity entity = response.getEntity();
132     if (entity != null) {
133       // copy stream to local file
134       OutputStream fos = new FileOutputStream(zip, false);
135       try {
136         entity.writeTo(fos);
137       } finally {
138         fos.close();
139       }
140     }
141 
142     LOG.info("Successfully downloaded {} to {}", url, zip.getAbsolutePath());
143 
144     // open archive
145     Archive arch = ArchiveFactory.openArchive(zip, dwca);
146     return arch;
147   }
