13  public class RsGbifOrg {
14    private static final Logger LOG = LoggerFactory.getLogger(RsGbifOrg.class);
15    private static final Joiner PATH_JOINER = Joiner.on("/").skipNulls();
16    private static final String DICT_DIR = "dictionaries";
17    public static final String DOMAIN = "http://rs.gbif.org/";
18    public static final String FILENAME_BLACKLIST = "blacklisted.txt";
19  
20    /**
21     * @param path given as array of individual names that will be concatenated
22     * @return url to file inside rs.gbif.org
23     */
24    public static URL url(String ... path) {
25      try {
26        if (path == null){
27          return new URL(DOMAIN);
28        }
29        return new URL(DOMAIN + PATH_JOINER.join(path));
30      } catch (MalformedURLException e) {
31        LOG.error("Cannot create rs.gbif.org url for path " + PATH_JOINER.join(path), e);
32      }
33      return null;
34    }
35  
36    /**
37    * @param filename of dictionary file requested
38    * @return url to file inside to dictionary folder of rs.gbif.org
39    */
40    public static URL dictionaryUrl(String filename) {
41      return url(DICT_DIR, filename);
42    }
43  
44    /**
45     * @param filename of authority dictionary file requested
46     * @return url to file inside to authority folder of rs.gbif.org
47     */
48    public static URL authorityUrl(String filename) {
49      return url(DICT_DIR, "authority", filename);
50    }
