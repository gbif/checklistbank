24  /**
25   * Tools that generate all files for a new nub build.
26   * Should be able to share this class at some point with importing routines when we get there.
27   * The files generated are ready to be quickly inserted into teh clb database.

29  public class NubWriter {
30    private static final Logger LOG = LoggerFactory.getLogger(NubWriter.class);
31    private final File dir;
32    private final DatasetImportService importService;
33    private TabWriter nubRelWriter;
34    private UUID currentDatasetKey;
35    private Set<UUID> processedDatasets = Sets.newHashSet();

37    @Inject
38    public NubWriter(DatasetImportService importService, File reportDir) {
39      this.importService = importService;
40      this.dir = reportDir;
41    }

43    /**
44     * Starts a new set of source dataset files and closes the last one if it was still open.
45     */
46    public void startDataset(UUID datasetKey) {
47      currentDatasetKey = datasetKey;
48      closeQuietly(nubRelWriter);
49      nubRelWriter = open(nubRelFile(datasetKey));
50      processedDatasets.add(datasetKey);
51    }

53    private File nubRelFile(UUID datasetKey) {
54      return new File(dir, "nubkeys-"+datasetKey.toString()+".tab");
55    }
56  
57    public void close() {
58      closeQuietly(nubRelWriter);
59    }
60  
61    public void closeQuietly(TabWriter w) {
62      if (w != null) {
63        try {
64          w.close();
65        } catch (IOException e) {
66          LOG.warn("Cannot close tab writer", e);
67        }
68      }
69    }
70  
71    public void mapSource(int srcUsageKey, int nubUsageKey) {
72      Preconditions.checkNotNull("no dataset yet started", nubRelWriter);
73      try {
74        String[] row = {String.valueOf(srcUsageKey), String.valueOf(nubUsageKey)};
75        nubRelWriter.write(row);
76      } catch (IOException e) {
77        throw new IllegalStateException("Failed to write nub key map", e);
78      }
79    }
80  
81    /**
82     * Deletes any previous nub from checklistbank and inserts a new one based on the nub cache.
83     */
84    public void insertNubData(ChecklistCache nubCache) {
85      LOG.info("Write nub data to files in directory {}", dir);
86  
87      // delete old nub
88      importService.deleteDataset(Constants.NUB_DATASET_KEY);
89  
90      // insert new nub usage
91      importService.insertUsages(Constants.NUB_DATASET_KEY, nubCache.iterate());
92    }
93  
94    private TabWriter open(File f) {
95      try {
96        return new TabWriter(new FileOutputStream(f));
97      } catch (FileNotFoundException e) {
98        throw new IllegalStateException("Cannot open tab writer for file " + f.getAbsolutePath(), e);
99      }
100   }

102   /**
103    * Uses the mapped source files to update all nub relations for those datasets.
104    */
105   public void updateRelNub() {
106     for (UUID datasetKey : processedDatasets) {
107       try {
108         Map<Integer, Integer> relations = readRelFile(nubRelFile(datasetKey));
109         LOG.info("Updating dataset {} with {} nub relations", datasetKey, relations.size());
110         importService.insertNubRelations(datasetKey, relations);
111       } catch (FileNotFoundException e) {
112         LOG.error("Cannot read nub relation file for dataset {}. Warning, dataset relations are unchanged!", datasetKey);
113       }
114     }
115   }

117   private Map<Integer, Integer> readRelFile(File f) throws FileNotFoundException {
118     Map<Integer, Integer> rels = Maps.newHashMap();
119     LineIterator lines = FileUtils.getLineIterator(FileUtils.getInputStream(f));
120     while (lines.hasNext()) {
121       String[] parts = FileUtils.TAB_DELIMITED.split(lines.nextLine().trim());
122       rels.put(Integer.parseInt(parts[0]), Integer.parseInt(parts[1]));
123     }
124     return rels;
125   }
